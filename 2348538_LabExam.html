class epsilon_greedy:
    def __init__(self,n_strategies, epsilon = 0.1):
        self.n_strategies = n_strategies
        self.epsilon = epsilon
        self.successes = np.zeros(n_strategies)
        self.counts = np.zeros(n_strategies)

    def choose_strategy(self):
        if np.random.random() < self.epsilon:
            strategy = np.random.randint(self.n_strategies)
        else:
            strategy = np.argmax(self.successes / self.counts)
        return strategy

    def update(self, strategy, reward):
        self.successes[strategy] += reward
        self.counts[strategy] += 1
